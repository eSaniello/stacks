CREATE TABLE freelancers (
    freelancer_id int NOT NULL,
    type_id int,
    naam varchar,
    achternaam varchar,
    email varchar,
    mobiel int,
    adres varchar,
    woonplaats varchar,
    opleiding text,
    werkervaring text,
    PRIMARY KEY (freelancer_id),
    FOREIGN KEY (type_id)
    REFERENCES type_user(type_id)  
);
